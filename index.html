class FullStack {
    constructor(frontEnd, backEnd) {
        this.frontEnd = frontEnd;
        this.backEnd = backEnd;
    }
        
    get frontEnd() {
        return this.frontEnd;
    }
        
    set frontEnd(frontEnd) {
        this.frontEnd = frontEnd;
    }
        
    get backEnd() {
        return this.backEnd;
    }
        
    set backEnd() {
        this.backEnd = backEnd;
    }
}
        
const frontEnd = [
    "HTML",
    "CSS",
    "Bootstrap"
    "JavaScript ES6",
    "Web Components"
    "React.js",
];
        
const backEnd = [
    "Node.js",
    "Python",
    "MySQL"
];
        
const fullStack = new FullStack(frontEnd, backEnd);
console.log(fullStack.frontEnd);
console.log(fullStack.backEnd);
